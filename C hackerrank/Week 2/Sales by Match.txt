#include<stdio.h>
int main()
{
    int n,a[100],i,j,c,t1;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d ",&a[i]);

    }
    for(i=0;i<n-1;i++   )
    {
        for(j=0;j<n-i-1;j++)
        {
            if(a[j]>a[j+1])
            {
                t1=a[j];
                a[j]=a[j+1];
                a[j+1]=t1;
            }
        }
    }
   
    
    c=0;
    for(i=0;i<n-1;)
    {
        
            if(a[i]==a[i+1])
            {
                c++;
                i=i+2;
            }
            else
            i=i+1;
        }
        
   printf("%d",c);
   return 0; 
}

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int *arr;
    int n, a = 0;
    scanf("%d", &n);

    arr = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }

    for (int i = 0; i < n; i++)
    {

        for (int j = 0; j < n; j++)
        {

            if (arr[i] == arr[j])
            {
                a++;
            }
        }
        if (a == 1)
        {
            printf("%d", arr[i]);
        }
        a=0;
    }

    return 0;
}

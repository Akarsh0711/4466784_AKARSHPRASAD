#include<stdio.h>
int main()
{
    int n,i,j,d,m,sum,a[101],f=0,c;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    scanf("%d",&d);
    scanf("%d",&m);
    for(i=0;i<n;i++)
    {
        c=1;
        sum=a[i];
        for(j=i+1;j<n;j++)
        {
            if(c<m)
            {
                if(sum<d)
                {sum=sum+a[j];
                c++;
            }
            }
            if(sum>d)
            {
                break;
            }
        }
        if((sum==d)&&(c==m))
        {
            f++;
        }
    }
    printf("%d",f);
    return 0;
}
